```javascript

quickTest()                 // Quick test

// Testing
testSignup(name, email, phone, pass)
testLogin(email, pass)

// Database
await firebaseGetCollection('users')
await firebaseGetCollection('items')
await firebaseAddDocument('items', {title: 'Test'})

// Check User
firebaseGetCurrentUser()
```
